        function parseJSONC(str) {

            let isInsideString = false;

            let isInsideEscape = false;

            let result = '';

            for (let i = 0; i < str.length; i++) {

                const char = str[i];

                if (isInsideEscape) { result += char; isInsideEscape = false; continue; }

                if (char === '\\') { isInsideEscape = true; result += char; continue; }

                if (char === '"') { isInsideString = !isInsideString; result += char; continue; }

                if (!isInsideString) {

                    if (char === '/' && str[i+1] === '/') {

                        while (i < str.length && str[i] !== '\n') i++;

                        result += '\n'; continue;

                    }

                    if (char === '/' && str[i+1] === '*') {

                        i += 2;

                        while (i < str.length - 1 && !(str[i] === '*' && str[i+1] === '/')) i++;

                        i++; continue;

                    }

                }

                result += char;

            }

            try { return JSON.parse(result); } 

            catch (e) { throw new Error("JSON 解析失败。请检查原配置是否有漏掉的逗号或括号！\n报错信息: " + e.message); }

        }


        // 辅助函数：通过关键字符串模糊估算行号范围

        function findLineRange(fullString, searchKey, blockLinesLength) {

            const index = fullString.indexOf(searchKey);

            if (index === -1) return "--";

            const upToKeyword = fullString.substring(0, index);

            const startLine = upToKeyword.split('\n').length - 2; // 大概往上退2行到花括号

            const endLine = startLine + blockLinesLength;

            return `第 ${startLine} ~ ${endLine}`;

        }


        function generateConfig() {

            const inputConfigText = document.getElementById('inputConfig').value.trim();

            const ip = document.getElementById('proxyIp').value.trim();

            const port = parseInt(document.getElementById('proxyPort').value.trim());

            const user = document.getElementById('proxyUser').value.trim();

            const pass = document.getElementById('proxyPass').value.trim();

            const domainsText = document.getElementById('proxyDomains').value;


            if (!inputConfigText) { alert("请先在第一步粘贴原有的 Xray 配置！"); return; }

            if (!ip || !port) { alert("代理 IP 和 端口不能为空！"); return; }


            let config;

            try { config = parseJSONC(inputConfigText); } 

            catch (error) { alert(error.message); return; }


            const tagId = "Static-ISP-socks5";

            let staticIspOutbound = {

                "tag": tagId,

                "protocol": "socks",

                "settings": { "servers": [{ "address": ip, "port": port }] }

            };

            if (user && pass) {

                staticIspOutbound.settings.servers[0].users = [{ "user": user, "pass": pass }];

            }


            let domainList = domainsText.split('\n').map(d => d.trim()).filter(d => d !== '');

            let staticIspRoutingRule = {

                "type": "field",

                "outboundTag": tagId,

                "domain": domainList

            };


            if (!config.outbounds) config.outbounds = [];

            if (!config.routing) config.routing = { rules: [] };

            if (!config.routing.rules) config.routing.rules = [];


            config.outbounds = config.outbounds.filter(out => out.tag !== tagId);

            config.routing.rules = config.routing.rules.filter(rule => rule.outboundTag !== tagId);


            let insertIndex = config.outbounds.findIndex(out => out.tag === "direct") + 1;

            config.outbounds.splice(insertIndex > 0 ? insertIndex : 0, 0, staticIspOutbound);


            let ruleIndex = config.routing.rules.findIndex(rule => rule.outboundTag !== "api");

            config.routing.rules.splice(ruleIndex > -1 ? ruleIndex : 0, 0, staticIspRoutingRule);


            const finalJsonString = JSON.stringify(config, null, 2);

            document.getElementById('outputCode').textContent = finalJsonString;


            const lineCount = finalJsonString.split('\n').length;

            const lineNumbersHtml = Array.from({length: lineCount}, (_, i) => i + 1).join('\n');

            document.getElementById('lineNumbers').textContent = lineNumbersHtml;


            // 动态计算行号并展示底部提示栏

            const userLines = user && pass ? 10 : 8; // 出站代码大概的行数跨度

            const routeLines = domainList.length + 3; // 路由代码大概的行数跨度

            

            document.getElementById('lineOutbound').textContent = findLineRange(finalJsonString, `"tag": "${tagId}"`, userLines);

            document.getElementById('lineRouting').textContent = findLineRange(finalJsonString, `"outboundTag": "${tagId}"`, routeLines);

            

            const infoBox = document.getElementById('injectionInfo');

            infoBox.classList.remove('hidden');

            infoBox.classList.add('flex');

        }

                setTimeout(() => {

                    btn.innerHTML = originalHtml;

                    btn.classList.replace('bg-blue-600', 'bg-green-600');

                    btn.classList.replace('hover:bg-blue-700', 'hover:bg-green-700');

                }, 2000);

            }).catch(err => {

                const textArea = document.createElement("textarea");

                document.body.appendChild(textArea);

                textArea.select();

                try { document.execCommand('copy'); alert('复制成功'); } catch (err) { alert('复制失败，请手动全选'); }

                document.body.removeChild(textArea);

            });

        }
